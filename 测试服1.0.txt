function startDrag(cx,cy) {
if (isCollapsed) return;
const r = root.getBoundingClientRect();
dragOffX=cx-r.left; dragOffY=cy-r.top; isDragging=true;
root.style.right = window.parent.innerWidth - r.right + "px";
root.style.top = r.top + "px"; root.style.left = "auto";
}
function moveDrag(cx,cy) {
if (!isDragging) return;
let nr = window.parent.innerWidth - cx + dragOffX - root.offsetWidth;
let nt = cy - dragOffY;
nr = Math.max(0, Math.min(window.parent.innerWidth - root.offsetWidth, nr));
nt = Math.max(0, Math.min(window.parent.innerHeight - root.offsetHeight, nt));
root.style.right=nr+"px"; root.style.top=nt+"px";
syncCollapseBtn();
if (thoughtsOpen) posThoughts();
if (reviewOpen) posReview();
if (chatOpen) posChatPanel();
if (apiOpen) posApiPanel();
}

if (!isDragging) return;
isDragging=false; avatarEl._wasDragged=true;
setTimeout(()=>{avatarEl._wasDragged=false;},50);
}

/* FIX6: 时间分隔线 — 记录实际epoch，只在30min间隔后插入 */
let _lastMsgEpoch=0;
function fmtTimeDiv(date){
const h=date.getHours(),m=date.getMinutes();
const p=h<12?"上午":h<18?"下午":"晚上";
const hh=h>12?h-12:(h===0?12:h);
return p+" "+hh+":"+String(m).padStart(2,"0");
}

const div=D.createElement("div");div.className="cht-tdiv";div.textContent=fmtTimeDiv(new Date(epochMs));

}

/* ── 健壮 JSON 解析 ── */
function tryParseJSON(raw) {
  if (!raw) return null;
  let s = stripThink(raw).replace(/^```json\s*/i,'').replace(/^```\s*/,'').replace(/\s*```$/,'').trim();
  try { return JSON.parse(s); } catch(_) {}
  const start = s.indexOf('{'), end = s.lastIndexOf('}');
  if (start !== -1 && end > start) {
    try { return JSON.parse(s.slice(start, end + 1)); } catch(_) {
      try { return JSON.parse(s.slice(start, end + 1).replace(/,\s*([}\]])/g, '$1')); } catch(_) {}
    }
  }
  return null;
}
